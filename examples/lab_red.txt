network lab_red {

    subnet LAN 192.168.1.0/24

    device R1 router
    device SW1 switch
    device PC1 host
    device PC2 host

    interface R1  eth0 ip 192.168.1.1 mask 255.255.255.0
    interface PC1 eth0 ip 192.168.1.2 mask 255.255.255.0
    interface PC2 eth0 ip 192.168.1.3 mask 255.255.255.0

    connect R1.eth0  to SW1.port1 bandwidth 1000 Mbps latency 1 ms
    connect PC1.eth0 to SW1.port2 bandwidth 100  Mbps
    connect PC2.eth0 to SW1.port3 bandwidth 100  Mbps
}
